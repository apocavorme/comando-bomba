Bomba dagua:
    site:
        X botao para desligar e ligar bomba
        - mostrar o nível da caixa dagua
        - mostrar o nível da bateria da caixa dagua
    
    client:
        casa:
            X informar para controle da casa o estado da bomba

    server:
        X recebe comando de ligar e desligar do controle da casa
        T recebe o nível da caixa dagua
        T recebe o nível da bateria

Controle da casa:
    client:
        - envia pra bomba dagua o comando de ligar/desligar
    
    server:
        - recebe o estado da bomba, enviado pela Bomba Dagua
        - recebe o nivel da caixa dagua, enviado pela Caixa Dagua

Caixa Dagua:
    client:
        - envia o nivel da caixa dagua para Controle da Casa e para Bomba Dagua
        - envia o nivel da bateria para a Bomba Dagua
